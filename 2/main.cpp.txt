#include <iostream>
#include <iomanip>

int main() {
    double d1, d2;
    
    // Ввод диагоналей ромба
    std::cout << "Введите длину первой диагонали ромба (0 < d1 < 100): ";
    std::cin >> d1;
    
    std::cout << "Введите длину второй диагонали ромба (0 < d2 < 100): ";
    std::cin >> d2;
    
    // Проверка корректности введенных данных
    if (d1 <= 0 || d2 <= 0 || d1 >= 100 || d2 >= 100) {
        std::cout << "Ошибка: диагонали должны быть в диапазоне (0 < d1,d2 < 100)" << std::endl;
        return 1;
    }
    
    // Вычисление площади ромба
    double area = (d1 * d2) / 2.0;
    
    // Вывод результата с точностью до 2 знаков после запятой
    std::cout << std::fixed << std::setprecision(2);
    std::cout << "Площадь ромба: " << area << std::endl;
    
    return 0;
}
